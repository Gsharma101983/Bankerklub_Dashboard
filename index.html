// ============================================================ // BankersKlub Dashboard — Google Apps Script Web App // Code.gs | Upgraded Version with: // • Referral Partner Count (Pipeline Overview) // • Source Name column in Raw Data // • CORS headers for direct browser fetch // • Multi-tab support (Master, Bankers Onboarding, CP Onboarding, Lender Tagging) // ============================================================ function doGet(e) { var sheetId = '1zo85_oRIDLQsR--kCmwWSYBDgm-ZElHIkoC9PNaUB2k'; var ss = SpreadsheetApp.openById(sheetId); // ── Which tab? ────────────────────────────────────────── var tabName = e.parameter.tab || 'Master'; var sheet = ss.getSheetByName(tabName); if (!sheet) { return buildResponse(JSON.stringify({ error: 'Sheet tab not found: ' + tabName })); } var data = sheet.getDataRange().getValues(); var headers = data[0].map(function(h){ return String(h).trim(); }); var rows = []; for (var i = 1; i < data.length; i++) { var row = {}; var hasData = false; for (var j = 0; j < headers.length; j++) { var val = data[i][j]; // Convert Date objects to ISO strings if (val instanceof Date) { val = val.toISOString(); } row[headers[j]] = val; if (val !== '' && val !== null && val !== undefined) hasData = true; } if (hasData) rows.push(row); // skip fully blank rows } // ── Extra aggregates only for Master tab ──────────────── var aggregates = {}; if (tabName === 'Master') { aggregates = buildAggregates(headers, rows); } var result = { tab: tabName, headers: headers, rows: rows, totalRows: rows.length, aggregates: aggregates, lastUpdated: new Date().toISOString() }; return buildResponse(JSON.stringify(result)); } // ── Build pipeline aggregates (called for Master tab) ─────── function buildAggregates(headers, rows) { // ── Helper: find header index (case-insensitive, partial match) ── function col(keyword) { var kw = keyword.toLowerCase(); for (var k = 0; k < headers.length; k++) { if (headers[k].toLowerCase().indexOf(kw) !== -1) return headers[k]; } return null; } // ── Key column names (auto-detected) ──────────────────── var colLeadSource = col('lead source') || col('source name') || col('source'); var colMandateType = col('mandate type'); var colBankerName = col('banker name') || col('banker'); var colCPName = col('channel partner') || col('cp name') || col('cp'); var colReferralPartner = col('referral partner') || col('referral'); var colLender = col('lender'); var colDebtAmt = col('debt') || col('loan amount'); var colEquityAmt = col('equity'); var colCaseStatus = col('case status') || col('status'); var colSignoffDate = col('signoff') || col('mandate signoff'); var colLeadDate = col('lead date') || col('date'); var colRevPipeline = col('rev in pipeline') || col('revenue pipeline') || col('expected revenue'); var colRevExpected = col('revenue expected') || col('rev expected'); var colAdvance = col('advance'); var colRevCollected = col('revenue collected') || col('collected'); var colRefund = col('refund'); var colMandateStatus = col('mandate status'); // ── Counters ───────────────────────────────────────────── var totalCount = rows.length; var uniqueBankers = {}; var uniqueCPs = {}; var uniqueReferrals = {}; var lenderCases = 0; var debtTotal = 0; var equityTotal = 0; var tatDays = []; var revPipeline = 0, revExpected = 0; var advanceAmt = 0, revCollected = 0, refundAmt = 0; var mandateStatusDist = {}; var mandateTypeDist = {}; var sourceDist = {}; rows.forEach(function(r) { // Bankers if (colBankerName && r[colBankerName]) { String(r[colBankerName]).split(',').forEach(function(b){ var t = b.trim(); if (t) uniqueBankers[t] = true; }); } // Channel Partners if (colCPName && r[colCPName]) { String(r[colCPName]).split(',').forEach(function(c){ var t = c.trim(); if (t) uniqueCPs[t] = true; }); } // Referral Partners (NEW) ──────────────────────────── // Strategy 1: dedicated Referral Partner column if (colReferralPartner && r[colReferralPartner]) { String(r[colReferralPartner]).split(',').forEach(function(rp){ var t = rp.trim(); if (t) uniqueReferrals[t] = true; }); } // Strategy 2: Lead Source = "Referral" rows → count unique names from CP / Banker col if (colLeadSource && r[colLeadSource]) { var src = String(r[colLeadSource]).trim().toLowerCase(); if (src === 'referral' || src.indexOf('referral') !== -1) { // Use CP name or Banker name as the referral entity var entity = (colCPName && r[colCPName]) || (colBankerName && r[colBankerName]) || ''; if (entity) uniqueReferrals[String(entity).trim()] = true; } } // Lender if (colLender && r[colLender]) lenderCases++; // Amounts if (colDebtAmt) debtTotal += parseFloat(r[colDebtAmt]) || 0; if (colEquityAmt) equityTotal += parseFloat(r[colEquityAmt]) || 0; // Revenue if (colRevPipeline) revPipeline += parseFloat(r[colRevPipeline]) || 0; if (colRevExpected) revExpected += parseFloat(r[colRevExpected]) || 0; if (colAdvance) advanceAmt += parseFloat(r[colAdvance]) || 0; if (colRevCollected) revCollected += parseFloat(r[colRevCollected]) || 0; if (colRefund) refundAmt += parseFloat(r[colRefund]) || 0; // TAT if (colLeadDate && colSignoffDate && r[colLeadDate] && r[colSignoffDate]) { try { var d1 = new Date(r[colLeadDate]); var d2 = new Date(r[colSignoffDate]); var diff = Math.round((d2 - d1) / 86400000); if (!isNaN(diff) && diff >= 0) tatDays.push(diff); } catch(ex){} } // Mandate Status distribution if (colMandateStatus && r[colMandateStatus]) { var ms = String(r[colMandateStatus]).trim(); mandateStatusDist[ms] = (mandateStatusDist[ms] || 0) + 1; } // Mandate Type distribution if (colMandateType && r[colMandateType]) { var mt = String(r[colMandateType]).trim(); mandateTypeDist[mt] = (mandateTypeDist[mt] || 0) + 1; } // Source Name distribution (NEW) ──────────────────── if (colLeadSource && r[colLeadSource]) { var sn = String(r[colLeadSource]).trim(); if (sn) sourceDist[sn] = (sourceDist[sn] || 0) + 1; } }); var avgTAT = tatDays.length ? Math.round(tatDays.reduce(function(a,b){ return a+b; }, 0) / tatDays.length) : 0; return { totalCount: totalCount, bankerCount: Object.keys(uniqueBankers).length, cpCount: Object.keys(uniqueCPs).length, referralPartnerCount: Object.keys(uniqueReferrals).length, // ← NEW lenderCases: lenderCases, debtTotal: debtTotal, equityTotal: equityTotal, avgTAT: avgTAT, revPipeline: revPipeline, revExpected: revExpected, advanceAmt: advanceAmt, revCollected: revCollected, refundAmt: refundAmt, netRevenue: revCollected - refundAmt, mandateStatusDist: mandateStatusDist, mandateTypeDist: mandateTypeDist, sourceDist: sourceDist, // ← NEW detectedColumns: { // ← diagnostic leadSource: colLeadSource, referralPartner: colReferralPartner, banker: colBankerName, cp: colCPName, mandateStatus: colMandateStatus, mandateType: colMandateType, lender: colLender } }; } // ── Wrap response with CORS headers ───────────────────────── function buildResponse(jsonString) { return ContentService .createTextOutput(jsonString) .setMimeType(ContentService.MimeType.JSON); }
